\begin{frame}
  \frametitle{The element matrix}

  The global matrix $A$ is defined by
  \begin{equation*}
    A_{ij} = a(\phi_j, \phi_i)
  \end{equation*}

  \bigskip

  The \emph{element matrix} $A_T$ is defined by
  \begin{equation*}
    A_{T,ij} = a_T(\phi_j^T, \phi_i^T)
  \end{equation*}

\end{frame}
