\begin{frame}
\frametitle{An algebraic splitting scheme}
In addition to 
\begin{equation}
\label{algs3}
M (u^{n+1} - u^*) =  - \Delta t B (p^{n+1} - p^n) 
\end{equation}
using previous notation with $u^c$ and $\phi$ where  
we have 
\[
u^{n+1} = u^* + u^c 
\]
and 
\[
\phi = p^{n+1} - p^n  
\]

Since 
\[
B^T u^{n+1} = 0 
\]
we must have that 
\begin{equation}
\label{algs4}
B^T u^c = - B^T u^* 
\end{equation}
and insering \eqref{algs3} into \eqref{algs4} 
we obtain 
\[
B^T M^{-1} B \phi = - \frac{1}{\Delta t} B^T u^* 
\]

\end{frame}
