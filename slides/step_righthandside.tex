\begin{frame}[fragile]
  \frametitle{Step by step: defining the right-hand side}

  The right-hand side $f = - 6$ may be defined as follows:
  \begin{python}
f = Expression("-6.0", degree=0)
  \end{python}

  \bigskip

  or (more efficiently) as
  \begin{python}
f = Constant(-6.0)
  \end{python}

\end{frame}
