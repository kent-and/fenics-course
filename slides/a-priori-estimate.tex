  \frametitle{\emph{A priori} estimates}
    If $u \in H^{k+1}(\Omega)$ and $V_h = P^k(\mcT_h)$ then
    \begin{gather*}
      \tn u - u_h \tn \leqslant C h^{k} \| u \|_{\Omega, k+1} \\
      \| u - u_h \|_{L^2(\Omega)} \leqslant C h^{k+1} \| u \|_{\Omega, k+1}
    \end{gather*}
