\begin{frame}[fragile]
  \frametitle{Useful FEniCS tools (I)}
  Mixed elements:
  \vspace{-0.5cm}
  \begin{python}
V = VectorFunctionSpace(mesh, "Lagrange", 2)
Q = FunctionSpace(mesh, "Lagrange", 1)
W = V*Q
  \end{python}
 Defining functions, test and trial functions:
  \vspace{-0.5cm}
 \begin{python}
up = Function(W)
(u,p) = split(up)
 \end{python}
 Shortcut:
  \vspace{-0.5cm}
 \begin{python}
(u, p) = Functions(W)
# similar for test and trial functions
(u, p) = TrialFunctions(W)
(v, q) = TestFunctions(W)
 \end{python}
\end{frame}
