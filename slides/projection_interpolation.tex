\begin{frame}[fragile]
  \frametitle{Projection and interpolation}

  We need to project the initial value into $V_h$:
  \vspace{-1em}
  \begin{python}
u0 = project(g, V)
  \end{python}

  \bigskip

  We can also interpolate the initial value into $V_h$:
  \vspace{-1em}
  \begin{python}
u0 = interpolate(g, V)
  \end{python}

\end{frame}
