\begin{frame}[fragile]
  \frametitle{Computing $1 + 2 + \ldots + 100$ in Python}

\begin{python}
s = 0

for i in range(1, 101):
    s += i

print s
\end{python}

\end{frame}

\begin{frame}[fragile]
  \frametitle{Running the program}

\begin{bash}
$ python sum.py
5050
\end{bash}

\end{frame}

\begin{frame}[fragile]
  \frametitle{Performance in Python vs C++}

  Let's compute $\sum_{k=1}^N k$ for $N = 100,000,000$.

\begin{bash}
$ time python sum.py
5000000050000000

real	0m13.243s
\end{bash}

\begin{bash}
$ time ./sum
5000000050000000

real	0m0.287s
\end{bash}

\end{frame}
